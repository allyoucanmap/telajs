tela.ent()

Adds a new Entity 

var options = {
 c: [0.0, 1.0, 1.0], // color
 mod: tela.mod({ ty: 'CUB' }), // model (required)
 ma: 1, // material
 n: 'myCube', // name
 o: [0.0, 0.0, 0.0], // origin
 p: [0.0, 0.0, 0.0], // position
 r: [0.0, 0.0, 0.0], // rotation
 rs: ['rx', 'ry', 'rz'], // sort rotation
 s: [1.0, 1.0, 1.0], // scale
 sp: [1.0, 0.0, 1.0], // specular color
 va: 1.0 // value
};

tela.ent(options);

<script language="javascript" type="text/javascript">

 var count = 0;

 tela.set('tela');
 tela.resize();

 var cube = tela.ent({
   s: 2,
   c: [0.0, 0.6, 0.8],
   sp: [1.0, 0.0, 0.5],
   mod: tela.mod({ ty: 'CUB' })
 });

 tela.play(function () {
   cube.px(Math.sin(count));
   cube.py(Math.cos(count));

   cube.rx(cube.rx() + 1.0);
   cube.ry(cube.ry() + 1.0);

   cube.ma(Math.floor(count/10 % 10));

   count += 0.05;
 });

</script>
