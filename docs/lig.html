tela.lig()

Adds a new Light 

var options = {
 c: [1.0, 1.0, 1.0], // color
 p: [2.0, 2.0, 2.0], // position
};

tela.lig(options);

<script language="javascript" type="text/javascript">

 var count = 0;

 tela.set('tela');
 tela.resize();

 var light = tela.lig({
   c: [1.0, 0.7, 0.7],
   p: [0.0, 0.0, 2.0]
 });

 tela.ent({
   mod: tela.mod({ ty: 'REC' }),
   s: 7
 });

 tela.play(function () {
   light.px(Math.sin(count) * 2);
   light.py(Math.cos(count) * 2);

   light.cb((Math.cos(count) + 1) / 2);

   count += 0.05;
 });

</script>
